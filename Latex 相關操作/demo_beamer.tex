\section{array, pmatrix and cases}
\begin{frame}{array, pmatrix and cases}
	By array,
	$$ \left( \begin{array}{cccccc}
	1 & 2 & 3 & 1 & 2 & 3 \\
	4 & 5 & 6 & 4 & 5 & 6 \\
	7 & 8 & 9 & 7 & 8 & 9
	\end{array} \right) $$

	By pmatrix,
	$$ \begin{pmatrix}
	1 & 2 & 3 & 1 & 2 & 3 \\
	4 & 5 & 6 & 4 & 5 & 6 \\
	7 & 8 & 9 & 7 & 8 & 9
	\end{pmatrix}$$

	By array,
	$$|x| = \left\{ \begin{array}{rr}
	x,  & \mbox{if $x \geq 0$} \\
	-x, & \mbox{if $x < 0$}
	\end{array} \right.$$

	By cases,
	$$ |x|=
	\begin{cases}
	x,  & \mbox{if $x \geq 0$} \\
	-x, & \mbox{if $x < 0$}
	\end{cases}$$
\end{frame}
